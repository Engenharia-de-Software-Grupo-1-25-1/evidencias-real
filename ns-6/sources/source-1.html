 package com.sedinternet.services;
 
 import java.io.File;
 import jakarta.mail.internet.InternetAddress;
 import jakarta.mail.internet.MimeMessage;
 
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.core.io.FileSystemResource;
 import org.springframework.mail.javamail.JavaMailSender;
 import org.springframework.mail.javamail.MimeMessageHelper;
 import org.springframework.stereotype.Service;
 
 @Service
 public class EmailService {
 
     @Autowired
     private JavaMailSender mailSender;
 
     /**
      * Envia um e-mail com as informações fornecidas.
      *
      * @param descRemetente Nome do remetente a ser exibido
      * @param emailDestino  Destinatário(s) separados por vírgula ou ponto e vírgula
      * @param assuntoEmail  Assunto do e-mail
      * @param corpoHtml     Corpo do e-mail no formato HTML
      * @param anexo         Caminho do arquivo anexo (opcional, pode ser vazio ou null)
      * @return true se o e-mail for enviado com sucesso; false caso contrário
      */
     public boolean enviarEmail(String descRemetente, String emailDestino, String assuntoEmail, String corpoHtml, String anexo) {
         if (emailDestino != null && !emailDestino.trim().isEmpty()) {
             try {
                 // Cria a mensagem MIME
                 MimeMessage message = mailSender.createMimeMessage();
                 // Se houver anexo, o segundo parâmetro deve ser true (multipart)
                 boolean multipart = (anexo != null && !anexo.trim().isEmpty());
                 MimeMessageHelper helper = new MimeMessageHelper(message, multipart, "UTF-8");
 
                 // Define o remetente (from)
                 helper.setFrom(new InternetAddress("comercial@sedsoft.com.br", descRemetente));
 
                 // Processa os destinatários, tratando separadores ";" e ","
                 String[] destinatarios;
                 if (emailDestino.contains(";")) {
                     destinatarios = emailDestino.split(";");
                 } else if (emailDestino.contains(",")) {
                     destinatarios = emailDestino.split(",");
                 } else {
                     destinatarios = new String[]{ emailDestino };
                 }
                 // Remove espaços em branco de cada endereço
                 for (int i = 0; i < destinatarios.length; i++) {
                     destinatarios[i] = destinatarios[i].trim();
                 }
                 helper.setTo(destinatarios);
 
                 // Define o assunto e o corpo (HTML) do e-mail
                 helper.setSubject(assuntoEmail);
                 helper.setText(corpoHtml, true);
 
                 // Define cabeçalhos de prioridade alta
                 message.setHeader("X-Priority", "1");
                 message.setHeader("X-MSMail-Priority", "High");
                 message.setHeader("Importance", "High");
 
                 // Se houver anexo, adiciona-o
                 if (multipart) {
                     File file = new File(anexo);
                     if (file.exists()) {
                         FileSystemResource fileResource = new FileSystemResource(file);
                         helper.addAttachment(fileResource.getFilename(), fileResource);
                     } else {
                         System.err.println("Arquivo de anexo não encontrado: " + anexo);
                     }
                 }
 
                 // Envia o e-mail
                 mailSender.send(message);
                 return true;
             } catch (Exception e) {
                 e.printStackTrace();
             }
         }
         return false;
     }
 }
