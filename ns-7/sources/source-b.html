 package com.sedinternet.services.impl;
 
 import com.sedinternet.entities.Metodo;
 import com.sedinternet.repositories.MetodoRepository;
 import com.sedinternet.services.MetodoService;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.stereotype.Service;
 
 @Service
 public class MetodoServiceImpl extends GenericServiceImpl<Metodo, Long> implements MetodoService {
 
     private final MetodoRepository metodoRepository;
 
     @Autowired
     public MetodoServiceImpl(MetodoRepository metodoRepository) {
         this.genericRepository = metodoRepository;
         this.metodoRepository = metodoRepository;
     }
 
     public Metodo findMetodoByDisciplinaOrTurma(Integer discId, Integer turmaId) {
         return metodoRepository.findMetodoByDisciplinaOrTurma(discId, turmaId);
     }
 
 }
