 package com.sedinternet.services.impl;
 
 import com.sedinternet.dtos.AlunoMediaDTO;
 import com.sedinternet.dtos.FreqAlunoDTO;
 import com.sedinternet.entities.FreqAluno;
 import com.sedinternet.repositories.FreqAlunoRepository;
 import com.sedinternet.services.FreqAlunoService;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.stereotype.Service;
 
 import java.time.LocalDate;
 import java.util.ArrayList;
 import java.util.List;
 
 @Service
 public class FreqAlunoServiceImpl extends GenericServiceImpl<FreqAluno, Long> implements FreqAlunoService {
 
     private final FreqAlunoRepository freqAlunoRepository;
 
     @Autowired
     public FreqAlunoServiceImpl(FreqAlunoRepository freqAlunoRepository) {
         this.genericRepository = freqAlunoRepository;
         this.freqAlunoRepository = freqAlunoRepository;
     }
 
     public List<FreqAluno> getFaltasByDisciplinaAndTurma(Integer disciplinaId, Integer turmaId) {
         return freqAlunoRepository.findFaltasByDisciplinaAndTurma(disciplinaId, turmaId);
     }
 
     @Override
     public List<FreqAlunoDTO> findAlunosPorTurmaDisciplinaEData(Long turmaId, Long disciplinaId, LocalDate dataAtividade) {
         return freqAlunoRepository.findAlunosPorTurmaDisciplinaEData(turmaId, disciplinaId, dataAtividade);
     }
 
     @Override
     public List<AlunoMediaDTO> findAlunosMediaByTurmaDisciplinaAndIntervalo(Long turmaId, Long disciplinaId, LocalDate etapaInicio, LocalDate etapaFim) {
         List<Object[]> results = this.freqAlunoRepository.findAlunosMediaByTurmaDisciplinaAndIntervalo(turmaId, disciplinaId, etapaInicio, etapaFim);
         List<AlunoMediaDTO> alunoMediaDTOList = new ArrayList<>();
 
         for (Object[] result : results) {
             AlunoMediaDTO dto = new AlunoMediaDTO();
 
             dto.setMatriculaId(result[0] != null ? ((Number) result[0]).longValue() : null);
             dto.setNomeAluno((String) result[1]);
             dto.setSituacao((String) result[2]);
             dto.setFaltas((Long) result[3]);
 
             alunoMediaDTOList.add(dto);
         }
         return alunoMediaDTOList;
     }
 
     @Override
     public void deleteFrequenciaPorResumo(Long resumoId){
         this.freqAlunoRepository.deleteFrequenciaPorResumo(resumoId);
     }
 
 }
