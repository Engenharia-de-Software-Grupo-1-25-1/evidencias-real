 package com.sedinternet.controllers;
 
 import com.sedinternet.dtos.GradeCurricularDTO;
 import com.sedinternet.entities.GradeCurricular;
 import com.sedinternet.services.GradeCurricularService;
 import org.springframework.web.bind.annotation.*;
 
 import java.util.List;
 
 @RestController
 @RequestMapping("/grade-curricular")
 public class GradeCurricularController extends GenericController<GradeCurricular, Long>{
 
     private final GradeCurricularService gradeCurricularService;
 
     public GradeCurricularController(GradeCurricularService gradeCurricularService) {
         super(gradeCurricularService);
         this.gradeCurricularService = gradeCurricularService;
     }
 
     @GetMapping("/turma/{turmaId}/professor/{professorId}")
     public List<GradeCurricularDTO> getDisciplinasByTurma(@PathVariable Integer turmaId, @PathVariable Integer professorId) {
         return gradeCurricularService.getDisciplinasByTurmaId(turmaId, professorId);
     }
 
     @GetMapping("/turma/{turmaId}/disciplina/{disciplinaId}")
     public List<Object[]> getSubDisc(@PathVariable Integer turmaId, @PathVariable Integer disciplinaId) {
         return gradeCurricularService.getSubDisc(turmaId, disciplinaId);
     }
 }
