 package com.sedinternet.controllers;
 
 import com.sedinternet.entities.ResumoAtividade;
 import com.sedinternet.services.ResumoAtividadeService;
 import jakarta.transaction.Transactional;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.format.annotation.DateTimeFormat;
 import org.springframework.http.ResponseEntity;
 import org.springframework.web.bind.annotation.*;
 import org.springframework.web.bind.annotation.PathVariable;
 
 import java.time.LocalDate;
 import java.util.List;
 
 @RestController
 @RequestMapping("/resumo-atividades")
 public class ResumoAtividadeController extends GenericController<ResumoAtividade, Long> {
 
     private final ResumoAtividadeService resumoAtividadeService;
 
     public ResumoAtividadeController(ResumoAtividadeService resumoAtividadeService) {
         super(resumoAtividadeService);
         this.resumoAtividadeService = resumoAtividadeService;
     }
 
     @Transactional
     @Override
     @PostMapping
     public ResponseEntity<ResumoAtividade> create(@RequestBody ResumoAtividade resumo) {
         try {
             ResumoAtividade savedResumo = resumoAtividadeService.save(resumo);
 
             return ResponseEntity.ok(savedResumo);
 
         } catch (Exception e) {
             return ResponseEntity.badRequest().body(null);
         } 
     }
 
     @GetMapping("/turma/{turmaId}/disciplina/{disciplinaId}/data/{dtAtividade}")
     public List<ResumoAtividade> getResumoByTurmaDiscData(
             @PathVariable("turmaId") Integer turmaId,
             @PathVariable("disciplinaId") Integer disciplinaId,
             @PathVariable("dtAtividade") @DateTimeFormat(iso = DateTimeFormat.ISO.DATE) LocalDate dtAtividade) {
         return this.resumoAtividadeService.findResumoByTurmaDiscDateAtvd(turmaId, disciplinaId, dtAtividade);
     }
 
     @GetMapping("/turma/{turmaId}/disciplina/{disciplinaId}/etapa/{etapa}")
     public List<ResumoAtividade> getResumoAtivToPrint(
             @PathVariable("turmaId") Integer turmaId,
             @PathVariable("disciplinaId") Integer disciplinaId,
             @PathVariable("etapa") Integer etapa) {
         return this.resumoAtividadeService.findResumoAtivToPrint(turmaId, disciplinaId, etapa);
     }
 
     @DeleteMapping("/{id}")
     @Override
     public ResponseEntity<Void> delete(@PathVariable Long id) {
         if (!genericService.findById(id).isPresent()) {
             return ResponseEntity.notFound().build();
         }
         this.resumoAtividadeService.deleteResumoById(id);
         return ResponseEntity.noContent().build();
     }
 }
