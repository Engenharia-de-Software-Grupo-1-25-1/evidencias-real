 package com.sedinternet.controllers;
 
 import com.sedinternet.dtos.AlunoMediaDTO;
 import com.sedinternet.dtos.FreqAlunoDTO;
 import com.sedinternet.entities.FreqAluno;
 import com.sedinternet.services.FreqAlunoService;
 import jakarta.transaction.Transactional;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.http.ResponseEntity;
 import org.springframework.web.bind.annotation.*;
 import java.time.LocalDate;
 import java.util.List;
 
 @RestController
 @RequestMapping("/freq-aluno")
 public class FreqAlunoController extends GenericController<FreqAluno, Long> {
 
     private final FreqAlunoService freqAlunoService;
 
     public FreqAlunoController(FreqAlunoService freqAlunoService) {
         super(freqAlunoService);
         this.freqAlunoService = freqAlunoService;
     }
 
     @Transactional
     @Override
     @PostMapping
     public ResponseEntity<FreqAluno> create(@RequestBody FreqAluno freq) {
         try {
             FreqAluno savedFreq = freqAlunoService.save(freq);
 
             return ResponseEntity.ok(savedFreq);
 
         } catch (Exception e) {
             return ResponseEntity.badRequest().body(null);
         }
     }
 
     @GetMapping("/disciplina/{disciplinaId}/turma/{turmaId}")
     public List<FreqAluno> getFaltasByDisciplinaAndTurma(@PathVariable Integer disciplinaId, @PathVariable Integer turmaId) {
         return freqAlunoService.getFaltasByDisciplinaAndTurma(disciplinaId, turmaId);
     }
 
     @GetMapping("/disciplina/{disciplinaId}/turma/{turmaId}/data/{dataAtividade}")
     public List<FreqAlunoDTO> getListagemAlunosFreq(@PathVariable Long turmaId, @PathVariable Long disciplinaId, @PathVariable LocalDate dataAtividade) {
         return freqAlunoService.findAlunosPorTurmaDisciplinaEData(turmaId, disciplinaId, dataAtividade);
     }
 
     @GetMapping("/turma/{turmaId}/disciplina/{disciplinaId}/etapaInicio/{etapaInicio}/etapaFim/{etapaFim}")
     public ResponseEntity<List<AlunoMediaDTO>> findAlunosMediaByTurmaDisciplinaAndIntervalo(
             @PathVariable("turmaId") Long turmaId,
             @PathVariable("disciplinaId") Long disciplinaId,
             @PathVariable("etapaInicio") LocalDate etapaInicio,
             @PathVariable("etapaFim") LocalDate etapaFim) {
         try {
             List<AlunoMediaDTO> medias = freqAlunoService.findAlunosMediaByTurmaDisciplinaAndIntervalo(turmaId, disciplinaId, etapaInicio, etapaFim);
             if (medias.isEmpty()) {
                 return ResponseEntity.noContent().build();
             }
             return ResponseEntity.ok(medias);
         } catch (Exception e) {
             return ResponseEntity.badRequest().build();
         }
     }
 
 }
