 package com.sedinternet.controllers;
 
 import com.sedinternet.dtos.GradeProfessorDTO;
 import com.sedinternet.entities.GradeProfessor;
 import com.sedinternet.repositories.GradeProfessorRepository;
 import com.sedinternet.services.GradeProfessorService;
 import org.springframework.web.bind.annotation.GetMapping;
 import org.springframework.web.bind.annotation.PathVariable;
 import org.springframework.web.bind.annotation.RequestMapping;
 import org.springframework.web.bind.annotation.RestController;
 import java.util.List;
 
 @RestController
 @RequestMapping("/grade-professor")
 public class GradeProfessorController extends GenericController<GradeProfessor, Long> {
 
     private final GradeProfessorService gradeProfessorService;
 
     public GradeProfessorController(GradeProfessorService gradeProfessorService, GradeProfessorRepository gradeProfessorRepository) {
         super(gradeProfessorService);
         this.gradeProfessorService = gradeProfessorService;
     }
 
     @GetMapping("/professor/{professorId}")
     public List<GradeProfessorDTO> get(@PathVariable Integer professorId) {
         return gradeProfessorService.getGradeProfessorByProfessorId(professorId);
     }
 }
