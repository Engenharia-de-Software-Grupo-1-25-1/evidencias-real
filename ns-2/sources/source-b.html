 package com.sedinternet.controllers;
 
 import com.sedinternet.entities.Metodo;
 import com.sedinternet.services.MetodoService;
 import org.springframework.web.bind.annotation.*;
 import org.springframework.http.ResponseEntity;
 
 @RestController
 @RequestMapping("/metodos")
 public class MetodoController extends GenericController<Metodo, Long> {
 
     private final MetodoService metodoService;
 
     public MetodoController(MetodoService metodoService) {
         super(metodoService);
         this.metodoService = metodoService;
     }
 
     @GetMapping("/disciplina/{discId}/turma/{turmaId}")
     public ResponseEntity<Metodo> getMetodoByDisciplinaOrTurma(
             @PathVariable Integer discId,
             @PathVariable Integer turmaId) {
 
         Metodo metodo = metodoService.findMetodoByDisciplinaOrTurma(discId, turmaId);
         return metodo != null ? ResponseEntity.ok(metodo) : ResponseEntity.notFound().build();
     }
 }
