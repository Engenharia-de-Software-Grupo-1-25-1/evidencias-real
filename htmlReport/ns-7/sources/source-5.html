 package com.sedinternet.services.impl;
 
 import com.sedinternet.entities.GenericEntity;
 import com.sedinternet.repositories.GenericRepository;
 import com.sedinternet.services.GenericService;
 import org.springframework.beans.factory.annotation.Autowired;
 
 import java.util.List;
 import java.util.Optional;
 
 public class GenericServiceImpl<T extends GenericEntity, ID> implements GenericService<T, ID> {
 
     @Autowired
     protected GenericRepository<T> genericRepository;
 
     @Override
     public T save(T entity) {
         return genericRepository.save(entity);
     }
 
     @Override
     public void deleteById(ID id) {
         genericRepository.deleteById((Long) id);
     }
 
     @Override
     public Optional<T> findById(ID id) {
         return genericRepository.findById((Long) id);
     }
 
     @Override
     public List<T> findAll() {
         return genericRepository.findAll();
     }
 }
