 package com.sedinternet.services.impl;
 
 import com.sedinternet.dtos.DisciplinaProfessorDTO;
 import com.sedinternet.entities.Turma;
 import com.sedinternet.repositories.TurmaRepository;
 import com.sedinternet.services.TurmaService;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.stereotype.Service;
 
 import java.util.ArrayList;
 import java.util.List;
 
 @Service
 public class TurmaServiceImpl extends GenericServiceImpl<Turma, Long> implements TurmaService {
 
     private final TurmaRepository turmaRepository;
 
     @Autowired
     public TurmaServiceImpl(TurmaRepository turmaRepository) {
         this.genericRepository = turmaRepository;
         this.turmaRepository = turmaRepository;
     }
 
     @Override
     public List<DisciplinaProfessorDTO> findDiscByProf(Integer professorId) {
         List<Object[]> resultList = turmaRepository.findDiscByProf(professorId);
         return convertToDisciplinaProfessorDTOList(resultList);
     }
 
     private List<DisciplinaProfessorDTO> convertToDisciplinaProfessorDTOList(List<Object[]> resultList) {
         List<DisciplinaProfessorDTO> dtos = new ArrayList<>();
         for (Object[] obj : resultList) {
             DisciplinaProfessorDTO dto = new DisciplinaProfessorDTO();
             dto.setTurmaId((Integer) obj[0]);
             dto.setTurmaNome((String) obj[1]);
             dto.setMetodoId((null));
             dtos.add(dto);
         }
         return dtos;
     }
 }
