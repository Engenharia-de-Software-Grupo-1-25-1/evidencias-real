 package com.sedinternet.services.impl;
 
 import com.sedinternet.dtos.AulasPorDiaDTO;
 import com.sedinternet.entities.TurmaAulas;
 import com.sedinternet.repositories.TurmaAulasRepository;
 import com.sedinternet.services.TurmaAulasService;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.stereotype.Service;
 
 import java.util.ArrayList;
 import java.util.List;
 
 @Service
 public class TurmaAulasImpl extends GenericServiceImpl<TurmaAulas, Long> implements TurmaAulasService {
 
     private final TurmaAulasRepository turmaAulasRepository;
 
     @Autowired
     public TurmaAulasImpl(TurmaAulasRepository turmaAulasRepository) {
         this.genericRepository = turmaAulasRepository;
         this.turmaAulasRepository = turmaAulasRepository;
     }
 
     public List<AulasPorDiaDTO> findAulasPorProfessorTurmaDisciplina(Integer professorId, Integer turmaId, Integer disciplinaId) {
         List<Object[]> results = turmaAulasRepository.findAulasPorProfessorDisciplinaETurma(professorId, disciplinaId, turmaId);
         List<AulasPorDiaDTO> aulasPorDiaDTOList = new ArrayList<>();
 
         for (Object[] result : results) {
             AulasPorDiaDTO dto = new AulasPorDiaDTO();
             dto.setProfessorId((Integer) result[0]);
             dto.setDisciplinaId((Integer) result[1]);
             dto.setQuantidadeAulas(((Number) result[2]).longValue());
             dto.setDiaDaSemana((String) result[3]);
             aulasPorDiaDTOList.add(dto);
         }
 
         return aulasPorDiaDTOList;
     }
 }
