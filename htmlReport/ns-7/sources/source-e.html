 package com.sedinternet.services.impl;
 
 import com.sedinternet.entities.ResumoAtividade;
 import com.sedinternet.repositories.ResumoAtividadeRepository;
 import com.sedinternet.services.FreqAlunoService;
 import com.sedinternet.services.ResumoAtividadeService;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.http.HttpStatus;
 import org.springframework.http.ResponseEntity;
 import org.springframework.stereotype.Service;
 
 import java.time.LocalDate;
 import java.util.List;
 
 @Service
 public class ResumoAtividadeServicecImpl extends GenericServiceImpl<ResumoAtividade, Long> implements ResumoAtividadeService {
 
     private final ResumoAtividadeRepository resumoAtividadeRepository;
     private final FreqAlunoService freqAlunoService;
 
     @Autowired
     public ResumoAtividadeServicecImpl(ResumoAtividadeRepository resumoAtividadeRepository,  FreqAlunoService freqAlunoService) {
         this.genericRepository = resumoAtividadeRepository;
         this.resumoAtividadeRepository = resumoAtividadeRepository;
         this.freqAlunoService = freqAlunoService;
     }
 
     @Override
     public List<ResumoAtividade> findResumoByTurmaDiscDateAtvd(Integer turmaId, Integer disciplinaId, LocalDate dtAtividade){
         return this.resumoAtividadeRepository.findResumoByTurmaDiscDateAtvd(turmaId, disciplinaId, dtAtividade);
     }
 
     @Override
     public List<ResumoAtividade> findResumoAtivToPrint(Integer turmaId, Integer disciplinaId, Integer etapa) {
         return this.resumoAtividadeRepository.findResumoAtivToPrint(turmaId, disciplinaId, etapa);
     }
 
     @Override
     public ResponseEntity<Void> deleteResumoById(Long resumoId) {
         try {
             resumoAtividadeRepository.deleteById(resumoId);
             freqAlunoService.deleteFrequenciaPorResumo(resumoId);
             return ResponseEntity.noContent().build();
         } catch (Exception e) {
             return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).build();
         }
     }
 }
