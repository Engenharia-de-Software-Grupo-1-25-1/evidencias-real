 package com.sedinternet.entities;

 import lombok.*;
 import java.time.LocalDate;

 @Data
 @EqualsAndHashCode(callSuper = false)
 @Entity
 @Builder
 @NoArgsConstructor
 @AllArgsConstructor
 @Table(name = "freq_alunos")
 public class FreqAluno extends GenericEntity {

     private Long disciplinaId;

     private Long turmaId;
 
     @Column(name = "DT_FREQ")

     @Column(name = "RESUMO_AT_ID")
     private Long resumoId;
 
     @Column(name = "ACS_ALU_ID")
     private Long acsAlunoId;
 
     @Column(name = "USR_CAD")
     private Integer usuarioCadastroId;
 
     @Column(name = "DT_CAD", nullable = false, columnDefinition = "TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP")
     private LocalDateTime dataCadastro;
 
     @Column(name = "USR_ALT")
     private Integer usuarioAlteracaoId;
 
     @Column(name = "DT_ALT", nullable = false, columnDefinition = "TIMESTAMP NOT NULL DEFAULT '0000-00-00 00:00:00'")

     private Integer quantidadeFaltas;
 
     @Column(name = "SITE")
     private Short site;
 
     @Column(name = "DT_SITE", columnDefinition = "TIMESTAMP NOT NULL DEFAULT '0000-00-00 00:00:00'")
     private LocalDateTime dataSite;
 
     @Column(name = "HORA_AULA_ID")
     private Integer horaAulaId;
 
     @Column(name = "N_FALTA")
     private Short numeroFaltas;
 
     @Column(name = "GRADE_ID")

 }
