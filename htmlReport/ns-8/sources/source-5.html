 package com.sedinternet.repositories;
 
 import com.sedinternet.dtos.FreqAlunoDTO;
 import com.sedinternet.entities.FreqAluno;
 import jakarta.transaction.Transactional;
 import org.springframework.data.jpa.repository.Modifying;
 import org.springframework.data.jpa.repository.Query;
 import org.springframework.data.repository.query.Param;
 import org.springframework.stereotype.Repository;
 
 import java.time.LocalDate;
 import java.util.List;
 
 @Repository
 public interface FreqAlunoRepository extends GenericRepository<FreqAluno> {
 
     @Query("SELECT fa FROM FreqAluno fa WHERE fa.disciplinaId = :disciplinaId AND fa.turmaId = :turmaId")
     List<FreqAluno> findFaltasByDisciplinaAndTurma(@Param("disciplinaId") Integer disciplinaId, @Param("turmaId") Integer turmaId);
 
     @Query("SELECT DISTINCT new com.sedinternet.dtos.FreqAlunoDTO(m.id, fa.id, a.nome, ra.etapa, sm.situacao, ra.id, " +
             "CASE WHEN fa.numeroFaltas IS NULL THEN 0 ELSE fa.numeroFaltas END, ra.qtdAula, fa.dataCadastro) " +
             "FROM Matricula m " +
             "INNER JOIN Aluno a ON m.aluno.id = a.id " +
             "LEFT JOIN FreqAluno fa ON fa.matriculaId = m.id AND fa.dtFreq = :dataAtividade AND fa.disciplinaId = :disciplinaId " +
             "LEFT JOIN SitMat sm ON sm.id = m.sit " +
             "LEFT JOIN ResumoAtividade ra ON ra.turmaId = m.turma.id " +
             "WHERE m.turma.id = :turmaId AND m.sit != 8 AND ra.disciplinaId = :disciplinaId " +
             "AND ra.dtAtividade = :dataAtividade " +
             "ORDER BY a.nome")
     List<FreqAlunoDTO> findAlunosPorTurmaDisciplinaEData(
             @Param("turmaId") Long turmaId,
             @Param("disciplinaId") Long disciplinaId,
             @Param("dataAtividade") LocalDate dataAtividade);
 
     @Transactional
     @Modifying
     @Query(value = """
         DELETE FROM freq_alunos fa WHERE fa.RESUMO_AT_ID = :resumoId
         """, nativeQuery = true)
     void deleteFrequenciaPorResumo(@Param("resumoId") Long resumoId);
 
     @Query("SELECT DISTINCT m.id, a.nome, " +
             "sm.situacao, " +
             "COALESCE(SUM(fa.numeroFaltas), 0) " +
             "FROM Matricula m " +
             "INNER JOIN Aluno a ON m.aluno.id = a.id " +
             "LEFT JOIN FreqAluno fa ON fa.matriculaId = m.id AND fa.disciplinaId = :disciplinaId AND fa.dtFreq BETWEEN :etapaInicio AND :etapaFim " +
             "LEFT JOIN SitMat sm ON sm.id = m.sit " +
             "WHERE m.turma.id = :turmaId AND m.sit != 8 " +
             "GROUP BY m.id, a.nome, sm.situacao " +
             "ORDER BY a.nome")
     List<Object[]> findAlunosMediaByTurmaDisciplinaAndIntervalo(
             @Param("turmaId") Long turmaId,
             @Param("disciplinaId") Long disciplinaId,
             @Param("etapaInicio") LocalDate etapaInicio,
             @Param("etapaFim") LocalDate etapaFim
     );
 
 }
