     public static String cifrar(String texto, long chave) {
         texto = texto.trim();
         int tamTexto = texto.length();
         StringBuilder tmp = new StringBuilder();
         int tamMax = 0;
 
         if (tamTexto > 0) {
             if (tamTexto % 2 != 0) {
                 tamTexto++;
                 texto += " ";
             }
 
             int[] iArray = new int[tamTexto];
             for (int i = 0; i < tamTexto; i++) {
                 int y = texto.charAt(i) ^ (int) (chave + 81 + tamTexto);
                 iArray[i] = y;
                 chave += 13;
 
                 int tamLetra = Integer.toString(y).length();
                 if (tamLetra > tamMax) {
                     tamMax = tamLetra;
                 }
             }
 
             String formato = "%0" + tamMax + "d";
 
             for (int i = 0; i < tamTexto; i += 2) {
                 String letra1 = String.format(formato, iArray[i]);
                 String letra2 = String.format(formato, iArray[i + 1]);
 
                 for (int z = 0; z < tamMax; z++) {
                     tmp.append(letra1.charAt(z)).append(letra2.charAt(z));
                 }
             }
 
             tmp.append(tamMax);
             String dv = DVCifrar(tmp.toString(), 'C');
             tmp.insert(0, dv.charAt(0)).append(dv.charAt(1));
         }
         return tmp.toString();
     }
 
     public static String DVCifrar(String texto, char modo) {
         try {
             texto = texto.trim();
             int tamTexto = texto.length();
             long soma = 0;
 
             if (modo == 'C') {
                 for (int i = 0; i < tamTexto; i++) {
                     soma += Character.getNumericValue(texto.charAt(i));
                 }
 
                 int dv1 = (int) (soma % 10);
                 int dv2 = tamTexto % 11;
                 if (dv2 == 10) {
                     dv2 = 0;
                 }
 
                 return "" + dv1 + dv2;
             }
         } catch (Exception e) {
             return "";
         }
         return "";
     }
