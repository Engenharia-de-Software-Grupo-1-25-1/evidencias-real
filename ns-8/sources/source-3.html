 package com.sedinternet.repositories;
 
 import com.sedinternet.entities.Etapa;
 import org.springframework.data.jpa.repository.Query;
 import org.springframework.stereotype.Repository;
 
 import java.util.List;
 
 @Repository
 public interface EtapaRepository extends GenericRepository<Etapa> {
 
     @Query(value = """
             SELECT e.ETAPA_ID,
                    m.NOME_ETAPA,
                    e.ETAPA,
                    e.METODO_ID,
                    e.ANO_LETIVO,
                    COALESCE(e.DT_DIG_IN, e.DT_INICIO) AS DT_DIG_IN,
                    COALESCE(e.DT_DIG_FIN, e.DT_FIM) AS DT_DIG_FIN
             FROM etapas e
             JOIN metodos m ON m.METODO_ID = e.METODO_ID
             WHERE e.METODO_ID = :metodoId
             AND e.ANO_LETIVO = YEAR(CURDATE());
             """,
     nativeQuery = true)
     List<Etapa> findEtapaByMetodo(Integer metodoId);
 
 }
